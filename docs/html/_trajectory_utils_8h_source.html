252  template <class Traits, class InputIterator,std::size_t LatIdx,
253  std::size_t LonIdx, std::size_t TsIdx, std::size_t SpeedIdx>
254  void get_speeds(InputIterator first, InputIterator beyond){
255 
256  typename Traits::Norm norm;
257  typename Traits::GeoProjection ref(std::get<LatIdx>(*first),
258  std::get<LonIdx>(*first));
259  typename Traits::MovetkPoint p = get_point<Traits>(std::get<LatIdx>(*first),
260  std::get<LonIdx>(*first), ref);
261  typename Traits::MovetkPoint prev = p;
262  auto it = first;
263  while (++it != beyond - 1){
264  p = get_point<Traits>(std::get<LatIdx>(*it),
265  std::get<LonIdx>(*it), ref);
266  typename Traits::MovetkVector v = p - prev;
267  typename Traits::NT squared_length = norm(v);
268  typename Traits::NT length = norm ^1;
269  typename Traits::NT delta_t = std::get<TsIdx>(*it) - std::get<TsIdx>(*(it - 1));
270  typename Traits::NT speed_v = length / delta_t;
271  std::get<SpeedIdx>(*it) = speed_v;
272  prev = p;
273  }
274  }

285  template<class GeometryKernel, class InputIterator, class PointsIterator, class OutputIterator,
286  typename = movetk_core::requires_random_access_iterator<InputIterator>,
287  typename = movetk_core::requires_random_access_iterator<PointsIterator>,
288  typename = movetk_core::requires_output_iterator<OutputIterator>,
289  typename = movetk_core::requires_pair<typename InputIterator::value_type>,
290  typename = movetk_core::requires_movetk_point<GeometryKernel,
291  typename PointsIterator::value_type>,
292  typename = movetk_core::requires_NT<GeometryKernel,
293  typename OutputIterator::value_type >>
294  void get_headings(InputIterator first, InputIterator beyond,
295  PointsIterator pfirst, OutputIterator iter) {
296  //ASSERT_RANDOM_ACCESS_ITERATOR(InputIterator);
297  //ASSERT_RANDOM_ACCESS_ITERATOR(PointsIterator);
298  // formula from: https://www.movable-type.co.uk/scripts/latlong.html
299  //TODO use default implementation of Geographiclib
300  typedef typename GeometryKernel::NT NT;
301  movetk_core::ComputeLength<GeometryKernel> distance;
302  InputIterator pit = first;
303  InputIterator cit = first + 1;
304  PointsIterator p_pit = pfirst;
305  PointsIterator c_pit = pfirst + 1;
306  *iter = 0;
307  iter++;
308  NT phi_1, lambda_1, phi_2, lambda_2;
309  NT x, y, c_heading, p_heading = 0;
310  while (cit != beyond) {
311  phi_1 = deg2radians(pit->first);
312  lambda_1 = deg2radians(pit->second);
313  phi_2 = deg2radians(cit->first);
314  lambda_2 = deg2radians(cit->second);
315  y = sin(lambda_2 - lambda_1) * cos(phi_2);
316  x = cos(phi_1) * sin(phi_2) - sin(phi_1) * cos(phi_2) * cos(lambda_2 - lambda_1);
317  c_heading = fmod(rad2deg(atan2(y, x)) + 360.0, 360.0);
318  if ((c_heading < MOVETK_EPS) && (distance(*p_pit, *c_pit) < MOVETK_EPS)) {
319  *iter = p_heading;
320  } else {
321  *iter = c_heading;
322  }
323  pit = cit;
324  p_pit = c_pit;
325  p_heading = c_heading;
326  cit++;
327  c_pit++;
328  }
329  }

458  typename = movetk_core::requires_NT<GeometryKernel,
459  typename InputIterator::value_type::second_type> >
460  InputIterator merge_intervals(InputIterator first, InputIterator beyond, bool sorted = false){
461  if (!sorted)
462  std::sort(first, beyond, [](auto a, auto b){
463  if (a.first != b.first)
464  return a.first > b.first;
465  else
466  return a.second > b.second;
467  });
468 
469  InputIterator curr = first;
470  InputIterator next = curr + 1;
471  InputIterator last = beyond - 1;
472  std::size_t moves = 0;
473  while (curr != last){
474  if (next->second >= curr->first){
475  moves++;
476  if (curr->second > next->second)
477  *curr = std::make_pair(next->first, curr->second);
478  else
479  *curr = std::make_pair(next->first, next->second);
480  *next = std::move(*(next + moves));
481  last--;
482  }
483  else{
484  curr++;
485  next++;
486  if ( moves > 0)
487  *next = std::move(*(next + moves));
488  }
489  }
490  return (last + 1);
491  }
