213  class TrajectoryDuration
214  {
215  using ItAccessor = IteratorAccessor<Trajectory>;
216  public:
217  using Time_t = typename Trajectory::template FieldType<TimeStampIdx>;
218  // Duration is the difference of the time type.

230  Duration_t operator()(Trajectory trajectory) const
231  {
232  // Setup the iterators
233  ItAccessor current = ItAccessor::start(trajectory);
234  ItAccessor end = ItAccessor::end(trajectory);
235 
236  // Minimum and maximum time
237  Time_t maxTime;
238  Time_t minTime = maxTime = current.template getField<TimeStampIdx>();
239 
240  // Find min and max time over all points
241  for (; current != end; ++current)
242  {
243  auto val = current.template getField<TimeStampIdx>();
244  if (minTime > val) minTime = val;
245  if (maxTime < val) maxTime = val;
246  }
247 
248  return maxTime - minTime;
249  }

252  template<typename Trajectory, typename GeometryKernel, class PointToPointDistance, int XcoordIdx, int YcoordIdx, int TimeStampIdx>
253  class TrajectorySpeedStatistic
254  {
255  public:
256  enum class Statistic
257  {
258  Median,
259  Mean,
260  Max,
261  Min,
262  Variance
263  };
264  // Time type
265  using Time_t = typename Trajectory::template FieldType<TimeStampIdx>;
266  // Duration is the difference of the time type.
267  using Duration_t = decltype(std::declval<Time_t>() - std::declval<Time_t>());
268  // Point type
269  using Point_t = typename GeometryKernel::MovetkPoint;
270  // Distance type
271  using Dist_t = decltype(std::declval<PointToPointDistance>()(std::declval<Point_t>(), std::declval<Point_t>()));
272  // Speed type = Distance type / Duration type
273  using Speed_t = decltype(std::declval<Duration_t>() / std::declval<Dist_t>());
274  private:
275  using ItAccessor = IteratorAccessor<Trajectory>;
276  // The distance function
277  PointToPointDistance m_dist;
278

288  return make_point({ it.template getField<XcoordIdx>(), it.template getField<YcoordIdx>() });
289  }

297  Speed_t getSpeed(ItAccessor it)
298  {
299  ItAccessor next = it + 1;
300  auto timeDiff = next.template getField<TimeStampIdx>() - it.template getField<TimeStampIdx>();
301  // Calculate the distance
302  auto dist = m_dist(getPoint(it), getPoint(next));
303  // Return the speed
304  return dist / timeDiff;

309  m_dist(dist){}

311  std::vector<Speed_t> operator()(Trajectory trajectory, const std::vector<Statistic>& requiredStatistics)
312  {
313  std::vector<Speed_t> speeds;
314  movetk_core::MakePoint<GeometryKernel> make_point;
315 
316  ItAccessor current = ItAccessor::start(trajectory);
317  ItAccessor end = ItAccessor::end(trajectory);
318 
319  // Find min and max time over all points
320  for (; (current + 1) != end; ++current)
321  {
322  speeds.push_back(getSpeed(current));
323  }
324  std::vector<Speed_t> output(requiredStatistics.size(), 0);
325 
326  // Check if the speeds are empty to avoid divisions by zero
327  if (speeds.empty()) return output;
328 
329  // Calculate the statistics
330  for(int i = 0; i < requiredStatistics.size(); ++i)
331  {
332  switch (requiredStatistics[i])
333  {
334  case Statistic::Median:
335  {
336  std::nth_element(speeds.begin(), speeds.begin() + speeds.size() / 2, speeds.end());
337  // Odd size
338  if (speeds.size() & 1)
339  {
340  output[i] = speeds[speeds.size() / 2];
341  }
342  else
343  {
344  // Even size
345  output[i] = 0.5 * (speeds[speeds.size() / 2] + speeds[speeds.size() / 2 - 1]);
346  }
347  break;
348  }
349  case Statistic::Mean:
350  {
351  // Compute the average
352  output[i] = std::accumulate(speeds.begin(), speeds.end(), static_cast<Speed_t>(0)) / static_cast<Speed_t>(speeds.size());
353  break;
354  }
355  case Statistic::Min:
356  {
357  output[i] = *std::min_element(speeds.begin(), speeds.end());
358  break;
359  }
360  case Statistic::Variance:
361  {
362  auto totalNum = static_cast<Speed_t>(speeds.size());
363  auto mean = std::accumulate(speeds.begin(), speeds.end(), static_cast<Speed_t>(0)) / totalNum;
364  // Lambda for accumulating squares of differences to mean.
365  auto sqDiff = [&mean](Speed_t accum, Speed_t newVal)
366  {
367  return accum + (newVal - mean)*(newVal - mean);
368  };
369  output[i] = std::accumulate(speeds.begin(), speeds.end(), static_cast<Speed_t>(0), sqDiff) / totalNum;
370  break;
371  }
372  case Statistic::Max:
373  {
374  output[i] = *std::max_element(speeds.begin(), speeds.end());
375  break;
376  }
377  }
378  }
379  return output;
380  }

389  template<typename Trajectory, int TimeStampIdx>
390  class TrajectoryTimeIntervalMode
391  {
392  using ItAccessor = IteratorAccessor<Trajectory>;
393  // Number of bins to use when computing mode on double precision durations
394  int m_binNum;
395  public:
396  using Time_t = typename Trajectory::template FieldType<TimeStampIdx>;
397  // Duration is the difference of the time type.
398  using Duration_t = decltype(std::declval<Time_t>() - std::declval<Time_t>());
399 
404  TrajectoryTimeIntervalMode(int binNum = 5): m_binNum(binNum){}
405 
418  Duration_t operator()(Trajectory trajectory)
419  {
420  std::vector<Duration_t> intervals;
421  ItAccessor current = ItAccessor::start(trajectory);
422  ItAccessor end = ItAccessor::end(trajectory);
423  // Determine all time intervals
424  for(; (current + 1) != end; ++current)
425  {
426  auto next = current + 1;
427  auto timeDiff = next.template getField<TimeStampIdx>() - current.template getField<TimeStampIdx>();
428  intervals.push_back(timeDiff);
429  }
430  // Sort the intervals
431  std::sort(intervals.begin(), intervals.end());
432 
433  // With integral types, we count occurences
434  if constexpr(std::is_integral_v<Duration_t>){
435  // Count occurences of the intervals
436  int occurences = 0;
437  int maxOccurences = 0;
438  Duration_t maxOccuredValue, prevValue;
439  bool isFirst = true;
440  for (auto val : intervals)
441  {
442  if (isFirst || val != prevValue)
443  {
444  prevValue = val;
445  occurences = 1;
446  }
447  else
448  {
449  occurences++;
450  }
451  isFirst = false;
452  if (maxOccurences < occurences)
453  {
454  maxOccurences = occurences;
455  maxOccuredValue = val;
456  }
457  }
458  return maxOccuredValue;
459  }
460  // With double types, use actual bins instead.
461  else
462  {
463  // Assumes that subtraction/addition and multiplication on
464  // the durations is defined
465  assert(m_binNum > 1);
466 
467  // Get min and max interval sizes
468  auto minInt = intervals[0];
469  auto maxInt = intervals[intervals.size() - 1];
470  auto diff = maxInt - minInt;
471 
472  // Check explicitly that not all values are the same
473  if (diff == 0) return intervals[0]; // This works for doubles since exact zero exists
474 
475  // Determine bin width. The bins are assigned
476  // such that the min and max value are the centers
477  // of the outer bins.
478  auto binWidth = diff / (m_binNum - 1);
479  auto start = minInt - 0.5 * binWidth;
480  //Setup the bins
481  std::vector<int> binCounts(m_binNum, 0);
482  // Count
483  for(auto el : intervals)
484  {
485  const int bin = static_cast<int>((el - start) / binWidth);
486  ++binCounts[bin];
487  }
488  // Return center of bin with maximum number of elements
489  const auto maxEl = std::max_element(binCounts.begin(), binCounts.end());
490  const int maxBin = static_cast<int>(maxEl - binCounts.begin());
491  return start + (maxBin + 0.5) * binWidth;
492  }
493  }
494  };
495 }
