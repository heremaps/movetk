38  class LongestCommonSubSequence

59  template <class InputIterator, class OutputIterator,
60  typename = movetk_core::requires_random_access_iterator<InputIterator>,
61  typename = movetk_core::requires_output_iterator<OutputIterator>,
62  typename = movetk_core::requires_movetk_point<GeometryTraits,
63  typename InputIterator::value_type>,
64  typename = movetk_core::requires_pair<typename OutputIterator::value_type>,
65  typename = movetk_core::requires_equality<typename InputIterator::value_type,
66  typename OutputIterator::value_type::first_type::value_type>,
67  typename = movetk_core::requires_equality<typename InputIterator::value_type,
68  typename OutputIterator::value_type::second_type::value_type>>
69  std::size_t operator()(InputIterator polyline_a_first, InputIterator polyline_a_beyond,
70  InputIterator polyline_b_first, InputIterator polyline_b_beyond, OutputIterator result)
71  {
72 
73  std::size_t size_polyline_b = std::distance(polyline_b_first, polyline_b_beyond);
74  std::vector<std::size_t> dp_row(size_polyline_b + 1);
75  std::fill(std::begin(dp_row), std::begin(dp_row) + size_polyline_b + 1, 0);
76  InputIterator it_a = polyline_a_first;
77  std::size_t i = 1, prev_value = 0, prev_cell = 0;
78  Norm norm;
79  while (it_a != polyline_a_beyond)
80  {
81  std::size_t j = 1, previous = 0, current = 0;
82  InputIterator it_b = polyline_b_first;
83  while (it_b != polyline_b_beyond)
84  {
85  typename GeometryTraits::MovetkVector v = *it_a - *it_b;
86  NT distance = norm(v);
87  if (distance < eps && this->abs(i, j) < del)
88  {
89  current = dp_row[j - 1] + 1;
90  if (current != prev_value)
91  {
92  prev_value = current;
93  prev_cell = j;
94  *result = std::make_pair(it_a, it_b);
95  }
96  else
97  {
98  if (j < prev_cell)
99  {
100  prev_cell = j;
101  *result -= std::make_pair(it_a, it_b);
102  }
103  }
104  }
105  else
106  current = std::max(dp_row[j], previous);
107  dp_row[j - 1] = previous;
108  previous = current;
109  j++;
110  it_b++;
111  }
112  dp_row[j - 1] = previous;
113  it_a++;
114  i++;
115  }
116  return dp_row.back();
117  }
118  };
