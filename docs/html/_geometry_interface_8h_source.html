370  template <class OutputIterator,
371  typename = movetk_core::requires_output_iterator<OutputIterator>,
372  typename = movetk_core::requires_tuple<
373  typename OutputIterator::value_type>,
374  typename = movetk_core::requires_L2_norm<typename IntersectionTraits::Norm>>
375  void operator()(typename IntersectionTraits::GeometryTraits::MovetkSphere &sphere,
376  typename IntersectionTraits::GeometryTraits::MovetkSegment &segment, OutputIterator result)
377  {
378  typename IntersectionTraits::Norm norm;
379  typedef typename IntersectionTraits::GeometryTraits::NT NT;
380  typedef typename IntersectionTraits::GeometryTraits::MovetkVector MovetkVector;
381  MovetkVector v1 = segment[1] - segment[0];
382  NT squared_length_v1 = norm(v1);
383  MovetkVector v2 = segment[0] - sphere.center();
384  NT squared_length_v2 = norm(v2);
385  NT q1 = std::pow(v1 * v2, 2) * (1 / squared_length_v1);
386  NT q2 = squared_length_v2 - sphere.squared_radius();
387  if (q1 < q2)
388  result = std::make_tuple(0, -1, -1.0, sphere.center());
389  else if ((q1 - q2) > (MOVETK_EPS * 1000))
390  {
391  NT n = 0.5;
392  NT root = std::pow(squared_length_v1 * (q1 - q2), n);
393  NT translation1 = -1 * (v1 * v2) + root;
394  NT translation2 = -1 * (v1 * v2) - root;
395  MovetkVector v3 = v1;
396  v3 *= (translation1 / squared_length_v1);
397  NT direction = v3 * v1;
398  NT length = norm(v3);
399  NT ratio = length / squared_length_v1;
400  if ((direction > 0) && (ratio <= 1))
401  {
402  result = std::make_tuple(0, 1, ratio, segment[0] + v3);
403  }
404  else if (direction < 0)
405  {
406  result = std::make_tuple(0, -1, ratio, segment[0]);
407  }
408  else if (ratio > 1)
409  {
410  result = std::make_tuple(0, -1, ratio, segment[1]);
411  }
412 
413  v3 *= (squared_length_v1 / translation1);
414  v3 *= (translation2 / squared_length_v1);
415  direction = v3 * v1;
416  length = norm(v3);
417  ratio = length / squared_length_v1;
418  if ((direction > 0) && (ratio <= 1))
419  {
420  result = std::make_tuple(0, 1, ratio, segment[0] + v3);
421  }
422  else if (direction < 0)
423  {
424  result = std::make_tuple(0, -1, ratio, segment[0]);
425  }
426  else if (ratio > 1)
427  {
428  result = std::make_tuple(0, -1, ratio, segment[1]);
429  }
430  }
431  else
432  {
433  NT translation = -1 * (v1 * v2);
434  MovetkVector v3 = v1;
435  v3 *= (translation / squared_length_v1);
436  NT direction = v3 * v1;
437  NT length = norm(v3);
438  NT ratio = length / squared_length_v1;
439  if ((direction > 0) && (ratio <= 1))
440  {
441  result = std::make_tuple(0, 0, ratio, segment[0] + v3);
442  }
443  }
444  }

446  template <typename = movetk_core::requires_L2_norm<typename IntersectionTraits::Norm>>
447  typename IntersectionTraits::GeometryTraits::MovetkSphere
448  operator()(typename IntersectionTraits::GeometryTraits::MovetkSphere &sphere_a,
449  typename IntersectionTraits::GeometryTraits::MovetkSphere &sphere_b)
450  {
451  // based on https://hal.archives-ouvertes.fr/hal-01955983/document
452  typename IntersectionTraits::Norm norm;
453  typedef typename IntersectionTraits::GeometryTraits GeometryTraits;
454  typedef typename GeometryTraits::NT NT;
455  typedef typename GeometryTraits::MovetkVector MovetkVector;
456  typedef typename GeometryTraits::MovetkPoint MovetkPoint;
457  movetk_core::MakePoint<GeometryTraits> make_point;
458  MakeSphere<GeometryTraits> make_sphere;
459  MovetkPoint ORIGIN = make_point({0, 0});
460  NT squared_r_a = sphere_a.squared_radius();
461  NT r_a = std::pow(squared_r_a, 0.5);
462  NT squared_r_b = sphere_b.squared_radius();
463  NT r_b = std::pow(squared_r_b, 0.5);
464  MovetkVector v = sphere_b.center() - sphere_a.center();
465  MovetkVector v1 = (sphere_b.center() - ORIGIN) +
466  (sphere_a.center() - ORIGIN);
467  NT squared_length = norm(v);
468  NT sum_squared_radius = squared_r_a + squared_r_b;
469  NT squared_diff_radius = sum_squared_radius - 2 * r_a * r_b;
470  NT squared_sum_radius = sum_squared_radius + 2 * r_a * r_b;
471  bool disjoint = (squared_sum_radius - squared_length < 0) || (squared_diff_radius - squared_length > 0);
472  if (!disjoint)
473  {
474  v *= (squared_r_a - squared_r_b) / (2 * squared_length);
475  v1 *= 0.5;
476  MovetkVector v2 = v1 + v;
477  MovetkPoint center = ORIGIN + v2;
478  NT op1 = (squared_r_a + squared_r_b) * 0.5;
479  NT numerator = (squared_r_a - squared_r_b) * (squared_r_a - squared_r_b);
480  NT op2 = 0.25 * numerator / squared_length;
481  NT squared_radius = op1 - op2 - 0.25 * squared_length;
482  return make_sphere(center, squared_radius, false);
483  }
484  else
485  {
486  return make_sphere(ORIGIN, 0);
487  }
488  }

549  template <class DistanceMatrix, class InputIterator,
550  typename = movetk_core::requires_random_access_iterator<InputIterator>,
551  typename = movetk_core::requires_random_access_iterator<typename InputIterator::value_type::iterator>,
552  typename = movetk_core::requires_movetk_point<GeometryTraits, typename InputIterator::value_type::value_type>,
553  typename = movetk_core::requires_random_access_iterator<typename DistanceMatrix::iterator>,
554  typename = movetk_core::requires_random_access_iterator<typename DistanceMatrix::value_type::iterator>,
555  typename = movetk_core::requires_NT<GeometryTraits, typename DistanceMatrix::value_type::value_type>>
556  typename GeometryTraits::NT
557  operator()(InputIterator first, InputIterator beyond, DistanceMatrix &upper_triag_mat)
558  {
559  typedef discrete_frechet_distance_algorithm<GeometryTraits, Norm,
560  typename GeometryTraits::MovetkDiscreteFrechetDistance>
561  algorithm;
562  typename algorithm::discrete_frechet_distance distance;
563  movetk_core::movetk_back_insert_iterator bit(upper_triag_mat);
564  auto it = first;
565  while (it != beyond)
566  {
567  auto nit = it + 1;
568  typename DistanceMatrix::value_type node;
569  movetk_core::movetk_back_insert_iterator node_bit(node);
570  while (nit != beyond)
571  {
572  node_bit = distance(std::cbegin(*it), std::cend(*it), std::cbegin(*nit), std::cend(*nit));
573  nit++;
574  }
575  bit = node;
576  it++;
577  }
578  auto rit = std::begin(upper_triag_mat);
579  typename GeometryTraits::NT min_sum = 1 / MOVETK_EPS;
580  std::size_t num_columns = std::distance(std::begin(*rit), std::end(*rit));
581  std::size_t row_id = 0;
582  while (rit != std::end(upper_triag_mat))
583  {
584  typename GeometryTraits::NT max = MOVETK_EPS;
585  typename GeometryTraits::NT second_max = MOVETK_EPS;
586  auto vit = std::begin(*rit);
587  std::size_t num_cols_right = std::distance(std::begin(*rit), std::end(*rit));
588  std::size_t num_cols_left = num_columns - num_cols_right;
589  std::size_t counter = 0;
590  while (counter != num_cols_left)
591  {
592  auto row_it = std::begin(upper_triag_mat) + counter;
593  std::size_t offset = row_id - counter - 1;
594  auto column_it = std::begin(*row_it) + offset;
595  if (*column_it > max)
596  {
597  second_max = max;
598  max = *column_it;
599  }
600  else if (*column_it > second_max && *column_it < max)
601  {
602  second_max = *column_it;
603  }
604  counter++;
605  }
606  while (vit != std::end(*rit))
607  {
608  if (*vit > max)
609  {
610  second_max = max;
611  max = *vit;
612  }
613  else if (*vit > second_max && *vit < max)
614  {
615  second_max = *vit;
616  }
617  vit++;
618  }
619  typename GeometryTraits::NT sum = max + second_max;
620  if (sum < min_sum)
621  {
622  min_sum = sum;
623  }
624  rit++;
625  row_id++;
626  }
627  return min_sum;
628  }
629  };

636  template <class GeometryTraits,
637  class Norm,
638  typename = movetk_core::requires_planar_geometry<GeometryTraits>,
639  typename = movetk_core::requires_L2_norm<Norm>>
640  class Wedge
641  {
642  private:
643  typedef typename GeometryTraits::NT NT;
644  Norm norm;
645  movetk_core::MakePoint<GeometryTraits> make_point;
646  typename GeometryTraits::MovetkPoint ORIGIN = make_point({0, 0});
647  typename GeometryTraits::MovetkVector e1 = make_point({1, 0}) - ORIGIN;
648  typename GeometryTraits::MovetkVector e2 = make_point({0, 1}) - ORIGIN;
649  typename GeometryTraits::MovetkVector _slope = ORIGIN - ORIGIN;
650  typename GeometryTraits::MovetkVector _intercept = ORIGIN - ORIGIN;
651  bool horizontal = false, vertical = false;
652  bool degenerate = false;
653  bool upper_right = false, lower_left = false;
654  bool lower_right = false, upper_left = false;
655 
656  public:
657  Wedge() = default;
658 
659  Wedge(typename GeometryTraits::MovetkPoint &p,
660  typename GeometryTraits::MovetkPoint &center,
661  typename GeometryTraits::NT radius)
662  {
663  NT m1, m2, c1, c2, tanA, tanB;
664  NT squared_radius = radius * radius;
665  typename GeometryTraits::MovetkVector _slope_ray = center - p;
666  NT v_x = _slope_ray * e1;
667  NT v_y = _slope_ray * e2;
668  NT segment_squared_length = norm(_slope_ray);
669  NT root = norm ^ 1;
670  if (segment_squared_length < squared_radius)
671  {
672  degenerate = true;
673  }
674  if (root < MOVETK_EPS)
675  {
676  degenerate = true;
677  }
678  if (abs(v_x) < MOVETK_EPS)
679  {
680  vertical = true;
681  }
682  if (abs(v_y) < MOVETK_EPS)
683  {
684  horizontal = true;
685  }
686  if (v_x >= 0 && v_y >= 0)
687  {
688  upper_right = true;
689  }
690  if (v_y <= 0 && v_x >= 0)
691  {
692  lower_right = true;
693  }
694  if (v_y >= 0 && v_x <= 0)
695  {
696  upper_left = true;
697  }
698  if (v_x <= 0 && v_y <= 0)
699  {
700  lower_left = true;
701  }
702  NT tangent_squared_length = segment_squared_length - squared_radius;
703  if (abs(tangent_squared_length) < MOVETK_EPS)
704  degenerate = true;
705  else
706  tanB = radius / sqrt(tangent_squared_length);
707  if (degenerate)
708  {
709  _slope = ORIGIN - ORIGIN;
710  }
711  else if (horizontal)
712  {
713  if (upper_right || lower_right)
714  _slope = make_point({tanB, -tanB}) - ORIGIN;
715  else if (upper_left || lower_left)
716  _slope = make_point({-tanB, tanB}) - ORIGIN;
717  }
718  else if (vertical)
719  {
720  tanA = tan(PI / 2 - MOVETK_EPS * 0.001);
721  if ((tanA * tanB) == 1)
722  m1 = (tanA + tanB) / MOVETK_EPS;
723  else
724  m1 = (tanA + tanB) / (1 - tanA * tanB);
725  m2 = -m1;
726  _slope = make_point({m1, m2}) - ORIGIN;
727  }
728  else
729  {
730  tanA = v_y / v_x; // slope of the ray
731  if ((tanA * tanB) == 1)
732  {

789  Wedge operator*(Wedge &w)
790  {
791  constexpr std::size_t size = 2 * Norm::P;
792  std::array<std::size_t, size> positions = {0, 1, 2, 3};
793  std::size_t sum = *(positions.end() - 1) + *(positions.end() - 2);
794  typename GeometryTraits::MovetkVector that_slope = w.slope();
795  typename GeometryTraits::MovetkVector that_intercept = w.intercept();
796  std::array<NT, size> slopes = {this->_slope * e1, this->_slope * e2,
797  that_slope * e1, that_slope * e2};
798  std::array<NT, size> intercepts = {this->_intercept * e1, this->_intercept * e2,
799  that_intercept * e1, that_intercept * e2};
800 
801  std::sort(std::begin(positions), std::end(positions), [&slopes](size_t i, size_t j) {
802  return slopes[i] < slopes[j];
803  });
804 
805  auto start = positions.cbegin();
806  auto end = positions.cend();
807  auto sit = slopes.cbegin();
808  auto it = intercepts.cbegin();
809 
810  NT diff_this = (slopes[0] - slopes[1]) / (1 + (slopes[0] * slopes[1]));
811  NT diff_that = (slopes[2] - slopes[3]) / (1 + (slopes[2] * slopes[3]));
812  NT WedgeAngle_this = 2 * movetk_core::rad2deg(diff_this);
813  NT WedgeAngle_that = 2 * movetk_core::rad2deg(diff_that);
814 
815  if ((*start + *(start + 1) == sum) || (*(end - 1) + *(end - 2) == sum))
816  {
817  if ((abs(WedgeAngle_this) < 90) || (abs(WedgeAngle_that) < 90))
818  return Wedge();
819  }
820 
821  if (slopes[0] > slopes[1])
822  {
823  that_slope = make_point({*(sit + *(start + 2)), *(sit + *(start + 1))}) - ORIGIN;
824  that_intercept = make_point({*(it + *(start + 2)), *(it + *(start + 1))}) - ORIGIN;
825  }
826  else
827  {
828  that_slope = make_point({*(sit + *(start + 1)), *(sit + *(start + 2))}) - ORIGIN;
829  that_intercept = make_point({*(it + *(start + 1)), *(it + *(start + 2))}) - ORIGIN;
830  }
831  return Wedge(that_slope, that_intercept);
832  }

842  NT mx2 = (make_point({this->_slope * this->e2, -1}) - ORIGIN) * (p - ORIGIN);
