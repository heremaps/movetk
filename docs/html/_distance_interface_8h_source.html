76  template <class DistanceMatrix, class InputIterator,
77  typename = movetk_core::requires_random_access_iterator<InputIterator>,
78  typename = movetk_core::requires_random_access_iterator<typename InputIterator::value_type::iterator>,
79  typename = movetk_core::requires_movetk_point<GeometryTraits, typename InputIterator::value_type::value_type>,
80  typename = movetk_core::requires_random_access_iterator<typename DistanceMatrix::iterator>,
81  typename = movetk_core::requires_random_access_iterator<typename DistanceMatrix::value_type::iterator>,
82  typename = movetk_core::requires_NT<GeometryTraits, typename DistanceMatrix::value_type::value_type>>
83  typename GeometryTraits::NT
84  operator()(InputIterator first, InputIterator beyond, DistanceMatrix &upper_triag_mat)
85  {
86  typedef discrete_frechet_distance_algorithm<GeometryTraits, Norm,
87  typename GeometryTraits::MovetkDiscreteFrechetDistance>
88  algorithm;
89  typename algorithm::discrete_frechet_distance distance;
90  movetk_core::movetk_back_insert_iterator bit(upper_triag_mat);
91  auto it = first;
92  while (it != beyond)
93  {
94  auto nit = it + 1;
95  typename DistanceMatrix::value_type node;
96  movetk_core::movetk_back_insert_iterator node_bit(node);
97  while (nit != beyond)
98  {
99  node_bit = distance(std::cbegin(*it), std::cend(*it), std::cbegin(*nit), std::cend(*nit));
100  nit++;
101  }
102  bit = node;
103  it++;
104  }
105  auto rit = std::begin(upper_triag_mat);
106  typename GeometryTraits::NT min_sum = 1 / MOVETK_EPS;
107  std::size_t num_columns = std::distance(std::begin(*rit), std::end(*rit));
108  std::size_t row_id = 0;
109  while (rit != std::end(upper_triag_mat))
110  {
111  typename GeometryTraits::NT max = MOVETK_EPS;
112  typename GeometryTraits::NT second_max = MOVETK_EPS;
113  auto vit = std::begin(*rit);
114  std::size_t num_cols_right = std::distance(std::begin(*rit), std::end(*rit));
115  std::size_t num_cols_left = num_columns - num_cols_right;
116  std::size_t counter = 0;
117  while (counter != num_cols_left)
118  {
119  auto row_it = std::begin(upper_triag_mat) + counter;
120  std::size_t offset = row_id - counter - 1;
121  auto column_it = std::begin(*row_it) + offset;
122  if (*column_it > max)
123  {
124  second_max = max;
125  max = *column_it;
126  }
127  else if (*column_it > second_max && *column_it < max)
128  {
129  second_max = *column_it;
130  }
131  counter++;
132  }
133  while (vit != std::end(*rit))
134  {
135  if (*vit > max)
136  {
137  second_max = max;
138  max = *vit;
139  }
140  else if (*vit > second_max && *vit < max)
141  {
142  second_max = *vit;
143  }
144  vit++;
145  }
146  typename GeometryTraits::NT sum = max + second_max;
147  if (sum < min_sum)
148  {
149  min_sum = sum;
150  }
151  rit++;
152  row_id++;
153  }
154  return min_sum;
155  }
